DFS & BFS
// DFS Example
void dfs(int u) {
    vis[u] = true;
    for (int v : adj[u])
        if (!vis[v]) dfs(v);
}

// BFS Example
queue<int> q;
q.push(start);
vis[start] = true;
while (!q.empty()) {
    int u = q.front(); q.pop();
    for (int v : adj[u])
        if (!vis[v]) {
            vis[v] = true;
            q.push(v);
        }
}
